def sqt(number, precision):

    n = 0
    while 10**(2*n) < number:
        n+=1
    n = n-1
    a = number/10**(2*n)
    s0 = (-190/(a+20)+10)*10**n

    k = 0
    while True:
        s1 = s0 - (s0**2-number)/(2*s0)
        if abs(s1**2 - number) < precision:
            break 
        s0 =s1
    return s1

print("Square root of 64:",sqt(64, 10**(-5)))
